G - 求三角形面积

```c
#include <stdio.h>
#include <math.h>
int  main(){
  double  a, b, c, p, s;
  scanf ( "%lf %lf %lf", &a, &b, &c);
  if (a + b > c && a + c > b && b + c > a)
  {
    p = (a + b + c) / 2;
    s = sqrt(p * (p - a) * (p - b) * (p - c));
    printf ( "%.2lf\n", s);
  }
  else  printf ("Fail\n");
  return  0;
}
```

H - 间距

```c
#include <stdio.h>
#include <math.h>
int main(){
    float d;
    int x1, y1, x2, y2;
    scanf("%d %d %d %d", &x1, &y1, &x2, &y2);
    d = sqrt((float)(x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
    printf("%.2lf", d);
    return 0;
}
```
